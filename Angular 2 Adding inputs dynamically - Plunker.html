import { Component } from "@angular/core";
@Component({
  selector: "app",
  template: `
  <form #formRef="ngForm">
    <div *ngFor="let word2 of words2; let in=index" 
      class="col-sm-3">
        <div class="form-group">
          <input type="text" [(ngModel)]="words2[in]
            .valu" name="name{{in}}" 
            class="form-control" #name="ngModel" 
            required />
          <!--<p [hidden]="words2[in].valu">Field 
            is required</p>
          {{formRef.form.name1}}-->
        </div>
        <br />
    </div>
    <button [disabled]="!formRef.form.valid" (click)
      ="add()">Add input</button>
  </form>
  <br />
  <br />
  {{words2 | json}}
  `
})
export class AppComponent {
  words2 = [{valu: 'word1'}, {valu: 'word2'}, 
    {valu: 'word3'}, {valu: ''}];
  
  add() {
    this.words2.pop();
  }
}
